# INSTALLATION GUIDE (Step-by-Step)

# 1. Create and activate virtual environment
python -m venv .venv
.\.venv\Scripts\Activate.ps1  # Windows PowerShell
# source .venv/bin/activate    # Linux/Mac

# 2. Upgrade pip first
python -m pip install --upgrade pip

# 3. Install core dependencies (minimal)
pip install -r requirements-minimal.txt

# 4. Install ML dependencies (if needed)
pip install -r requirements-ml.txt

# 5. Download spaCy model (if using ML features)
python -m spacy download en_core_web_sm

# 6. Train the model (if using ML)
flask train-chatbot

# 7. Test the chatbot
flask test-chatbot

# 8. Run the application
python run.py

# PRODUCTION DEPLOYMENT
export FLASK_ENV=production
export SECRET_KEY=your-secret-key
gunicorn -w 4 -b 0.0.0.0:5000 app:app

# TROUBLESHOOTING
# If packages fail to install:
# 1. Try installing one by one: pip install flask python-dotenv
# 2. Use --no-cache-dir: pip install --no-cache-dir -r requirements.txt
# 3. Try specific versions: pip install Flask==2.3.3
# 4. Check Python version compatibility: python --version
